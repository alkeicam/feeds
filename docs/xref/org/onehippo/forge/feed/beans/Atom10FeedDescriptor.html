152     @Override
153     public String process(final Feed syndication) {
154         Writer writer = null;
155         String feed = null;
156         try {
157             writer = new StringWriter();
158             WireFeedOutput output = new WireFeedOutput();
159             output.output(syndication, writer);
160             feed = writer.toString();
161         } catch (FeedException | IOException e) {
162             log.error("", e);
163         } finally {
164             IOUtils.closeQuietly(writer);
165         }
166         return feed;
167     }
