63      @GET
64      @Produces(MediaType.APPLICATION_XML)
65      @Path("/")
66      public String getFeed(@Context HttpServletRequest request, @Context HttpServletResponse response, @Context UriInfo info) {
67          String feed = null;
68          try {
69              long indexCount = 0;
70              long limit = 10L;
71              String sortByField;
72              HstRequestContext requestContext = getRequestContext(request);
73              HippoDocumentBean contentBean = getRequestContentBean(requestContext, HippoDocumentBean.class);
74  
75              if (contentBean == null || !(contentBean instanceof FeedDescriptor)) {
76                  log.error("Cannot find feed descriptor. Check relative content path on feed's sitemap item.");
77                  return null;
78              }
79              @SuppressWarnings("unchecked")
80              FeedDescriptor<T, E> document = (FeedDescriptor<T,E>) contentBean;
81              final T channel = ConversionUtil.covertToAppropriateSyndicationFeed(document, document, this, requestContext);
82  
83              // prevent NPE from auto-deboxing
84              if (document.getItemCount() != null) {
85                  limit = document.getItemCount();
86              }
87              sortByField = document.getSortByField();
88  
89              // get scope for search query
90              HippoBean scopeBean = getMountContentBaseBean(requestContext);
91              String scope = document.getScope();
92              if (scope != null && !"".equals(scope)) {
93                  HippoBean folderBean = scopeBean.getBean(scope);
94                  if (folderBean != null) {
95                      scopeBean = folderBean;
96                  }
97              }
98  
99              // FIXME get date field from BE template?
100             HstQuery hstQuery = null;
101             final String[] documentTypes = DocumentTypeHelper.getDocTypes(getDocTypes(requestContext, document), getExcludedDocTypes(document));
102             if (!ArrayUtils.isEmpty(documentTypes)) {
103                 hstQuery = getHstQueryManager(requestContext.getSession(), requestContext)
104                         .createQuery(scopeBean.getNode(), true, documentTypes);
105             }
106             if (hstQuery == null) {
107                 // dunno if we should go further at this point?
108                 hstQuery = getHstQueryManager(requestContext.getSession(), requestContext).createQuery(scopeBean);
109             }
110             if (sortByField != null) {
111                 hstQuery.addOrderByDescending(sortByField);
112             }
113 
114 
115             hstQuery.setLimit((int) limit);
116             if (modifier != null) {
117                 modifier.modifyHstQuery(requestContext, hstQuery, document);
118             }
119             HstQueryResult queryResult = hstQuery.execute();
120             HippoBeanIterator beans = queryResult.getHippoBeans();
121 
122             List<E> entries = new ArrayList<>();
123             while (beans.hasNext() && indexCount < limit) {
124                 HippoBean bean = beans.nextHippoBean();
125                 if (bean != null) {
126                     final E item = ConversionUtil.covertToAppropriateSyndicationEntry(document, bean, this, requestContext);
127                     if (modifier != null) {
128                         modifier.modifyEntry(requestContext, item, bean);
129                     }
130 
131                     entries.add(item);
132                 }
133                 indexCount++;
134             }
135             document.set(channel, entries);
136             if (modifier != null) {
137                 modifier.modifyFeed(requestContext, channel, document);
138             }
139             feed = document.process(channel);
140         } catch (Exception e) {
141             log.error("Cannot execute query for RSS feed. {e}", e);
142         }
143         return feed;
144     }

146     private Set<String> getExcludedDocTypes(final FeedDescriptor<T, E> document) {
147         Set<String> result = new HashSet<>();
148         if (document.getExclude()!=null){
149             String[] excludedDocumentTypes = StringUtils.split(document.getExclude(), ", \t\r\n");
150             result.addAll(Arrays.asList(excludedDocumentTypes));
151         }
152         return result;
153     }
154 
155     private Set<DocType> getDocTypes(final HstRequestContext requestContext, final FeedDescriptor<T, E> document) throws RepositoryException {
156         String [] documentTypes = StringUtils.split(document.getDocumentType(), ", \t\r\n");
157         NodeTypeManager nodeTypeManager = requestContext.getSession().getWorkspace().getNodeTypeManager();
158         Set<DocType> docTypes = new HashSet<>();
159         for (String documentType : documentTypes) {
160             NodeType nodeType = nodeTypeManager.getNodeType(documentType);
161             NodeTypeIterator iterator = nodeType.getSubtypes();
162             Set<String> subTypes = new HashSet<>();
163             while (iterator.hasNext()){
164                 subTypes.add(iterator.nextNodeType().getName());
165             }
166             docTypes.add(new DocType(documentType,subTypes));
167         }
168         return docTypes;
169     }
